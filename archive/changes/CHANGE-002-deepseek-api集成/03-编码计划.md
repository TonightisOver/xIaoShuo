```python
class Settings(BaseSettings):

    DEEPSEEK_MODEL: str = "deepseek-chat"
```

### 任务 4: 实现 LLM 客户端
**文件**: `src/core/llm/__init__.py`, `src/core/llm/client.py`

**关键代码**:
```python
from langchain_openai import ChatOpenAI
from tenacity import retry, stop_after_attempt, wait_exponential

class LLMClient:
    def __init__(self):
        settings = get_settings()
        self.llm = ChatOpenAI(
            model=settings.DEEPSEEK_MODEL,
            api_key=settings.DEEPSEEK_API_KEY,
            base_url=settings.DEEPSEEK_BASE_URL,
            temperature=settings.DEEPSEEK_TEMPERATURE,
            max_tokens=settings.DEEPSEEK_MAX_TOKENS,
            timeout=settings.DEEPSEEK_TIMEOUT,
        )
    
    @retry(stop=stop_after_attempt(3), wait=wait_exponential(multiplier=1, min=2, max=10))
    async def generate(self, prompt: str, **kwargs) -> str:
        messages = [{"role": "user", "content": prompt}]
        response = await self.llm.ainvoke(messages, **kwargs)
        return response.content
```

```python
from src.core.llm.client import get_llm_client
from src.core.llm.prompts import IDEA_EXPANSION_PROMPT

async def node(state: NovelState) -> NovelState:
    try:
        client = get_llm_client()
        prompt = IDEA_EXPANSION_PROMPT.format(
            idea=state["idea"],
            novel_type=state["novel_type"],
            target_words=state["target_words"]
        )
        expanded_idea = await client.generate(prompt)
        
        return {
            **state,
            "idea": expanded_idea,
            "current_stage": "idea_expansion_completed",
        }
    except Exception as e:
        # 降级到 mock 数据
        return {
            **state,
            "idea": state["idea"] + " (扩展后的创意 - mock)",
            "current_stage": "idea_expansion_completed",
            "errors": state["errors"] + [f"idea_expansion failed: {str(e)}"],
        }
```
